القائمة مثل حقيبة مدرسية نضع فيها عدة أشياء!

Une liste est comme un sac d'école où on met plusieurs choses!
notes = [85, 90, 78, 92, 88]    # درجات الطالب
matieres = ["math", "physique", "chimie"]  # المواد الدراسية
couleurs = ["rouge", "bleu", "vert"]       # الألوان

# الوصول لعنصر معين
premiere_note = notes[0]    # أول درجة (85)
derniere_note = notes[-1]   # آخر درجة (88)

print("أول درجة:", premiere_note)
print("كل الدرجات:", notes)

4الدوائر - Loops

Les Boucles

الدائرة تجعل الكمبيوتر يكرر نفس العمل عدة مرات!

Une boucle fait répéter le même travail à l'ordinateur plusieurs fois!
# طباعة الأرقام من 1 إلى 5
for nombre in range(1, 6):
    print("الرقم:", nombre)

# حساب مجموع القائمة
notes = [85, 90, 78, 92, 88]
somme = 0
for note in notes:
    somme = somme + note    # إضافة كل درجة للمجموع
    
moyenne = somme / len(notes)  # حساب المتوسط
print("المتوسط:", moyenne)

5الشروط - Conditions

Les Conditions

الشرط يجعل الكمبيوتر يقرر ماذا يفعل!

Une condition fait décider à l'ordinateur quoi faire!
note = 85

if note >= 90:
    message = "ممتاز! درجة عالية جداً"
elif note >= 80:
    message = "جيد جداً! استمر هكذا" 
elif note >= 70:
    message = "جيد، يمكنك التحسن"
else:
    message = "تحتاج للمزيد من المذاكرة"

print("تقييمك:", message)
